select Usuario.idUsuario as idUsuario, Adm.idAdm as idAdm, Elaborador.idElaborador as idElaborador, Respondedor.idRespondedor as idRespondedor

from Usuario

left join Adm on Usuario.idUsuario = Adm.idUsuario
left join Elaborador on Usuario.idUsuario = Elaborador.idUsuario
left join Respondedor on Usuario.idUsuario = Respondedor.idUsuario;
